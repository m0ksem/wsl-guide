# -------------- COLOR GIT -------------
PURPLE_START="\033[37;1;45m"
PURPLE_END="\033[0m"
NO_COLOR="\[\033[0m\]"

parse_git_branch() {
local git_branch=$(git branch 2> /dev/null | sed -e '/^[^*]/d' -e 's/* \(.*\)/(\1)/')
echo $git_branch
}
PS1="$PURPLE_START\$(parse_git_branch)$PURPLE_END$NO_COLOR${debian_chroot:+($debian_chroot)}\w$ "
